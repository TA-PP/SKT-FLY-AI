	num int not null,
	name varchar(10),
	address varchar(80),
	tel varchar(20),
	email varchar(20),
	primary key(num)
);

# 테이블 구조 보기

# 테이블 삭제

# 테이블 수정
create table friend (
	num int not null auto_Increment,
	name varchar(10),
	address varchar(80),
	tel varchar(20),
	email varchar(20),
	primary key(num)
);

# 필드 추가하기
